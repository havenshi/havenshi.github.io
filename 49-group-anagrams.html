49. Group Anagrams

Fri 06 January 2017
By Haven

In Python.

tags: pythoncode

class Solution(object):
    def groupAnagrams(self, strs):
        """
        :type strs: List[str]
        :rtype: List[List[str]]
        """
        map = {}
        for item in strs:
            order = ("".join(sorted(item))).lower()
            if order not in map:
                map[order] = [item.lower()]
            else:
                map[order] += [item.lower()]

        l = []
        for i in map.keys():
            l.append(map[i])
        return l
